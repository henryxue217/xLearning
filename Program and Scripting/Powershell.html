---Copy folder and files---

                                            
####Start####
###Copy files and folder is not existed or modified with copying process
function Copy-FolderIfNewer {
    param (
    [string]$SourcePath,
    [string]$DestinationPath
    )

# Get all items to copy (files and directories)
$itemsToCopy = Get-ChildItem -Path $SourcePath -Recurse 
$totalItems = $itemsToCopy.Count
$copiedItems = 0

Get-ChildItem -Path $SourcePath -Recurse | ForEach-Object {
    $copiedItems++
    $progressPercent = ($copiedItems / $totalItems) * 100
    $status = "Copying '$($_.Name)'..."

Write-Progress -Activity "Copying Items from '$SourcePath' to '$DestinationPath'" `
    -Status $status `
    -PercentComplete $progressPercent `
    -CurrentOperation "Copied $copiedItems of $totalItems items"


#remove the SourcePath string from an $_. and then add it to the destinationItemPath
$destinationItemPath = Join-Path -Path $DestinationPath -ChildPath $_.FullName.Substring($SourcePath.Length)

if ($_.PSIsContainer) {
    # It's a directory
    if (-not (Test-Path $destinationItemPath)) {
        New-Item -ItemType Directory -Path $destinationItemPath | Out-Null
        Write-Host "Created directory: $destinationItemPath"
        }
    } else {
    # It's a file
    if (-not (Test-Path $destinationItemPath) -or ($_.LastWriteTime -gt (Get-Item $destinationItemPath).LastWriteTime)) {

        Copy-Item -Path $_.FullName -Destination $destinationItemPath -Force
        Write-Host "Copied/Updated file: $($_.FullName) to $destinationItemPath"
        }
    }
    }

    Write-Progress -Activity "Copying Items from '$SourcePath' to '$DestinationPath'" `
    -Status "Copy Complete!" `
    -PercentComplete 100 `
    -Completed

}

####End####
